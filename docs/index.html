Common Utilities Package

A package of commonly used JavaScript utilities.

Installation
          npm i --save common-utils-pkg
Usage

ES6 Import

        import ColorUtil from 'common-utils-pkg';
          // if you want multiple utilities
          import { ArrayUtil, BoolUtil, ColorUtil } from 'common-utils-pkg';

          ColorUtil.randomPastelColor();
          // => "hsl(24.380295231593017,27.395530909987613%,86.86101287292395%)"

UNPKG

        
        <script src="https://unpkg.com/common-utils-pkg"></script>

            <script>
              const CUP = window['common-utils-pkg];
              CUP.ColorUtil.randomPastelColor();
              // => "hsl(24.380295231593017,27.395530909987613%,86.86101287292395%)"

              /* Or specify utility */
              const ColorUtil = window['common-utils-pkg'].ColorUtil;
              ColorUtil.randomPastelColor();
            </script>
